\begin{frame}{6. Pré-processamento e complexidade}
    \framesubtitle{Pipeline de dados unificado (\texttt{clean\_data.py})}
    \begin{enumerate}
        \item \textbf{Merge dos dados:} \\
        União das tabelas de metadados e sinopses via \textit{Hash Join} interno do Pandas. \\
        \textit{Complexidade:} $\mathcal{O}(N)$ no caso médio.
        
        \item \textbf{Parse de gêneros:} \\
        Extração de strings em formato JSON do Freebase $\rightarrow$ convertido para strings simples separadas por ``;``. \\
        \textit{Complexidade:} $\mathcal{O}(N \cdot g)$, onde $g$ é a quantidade média de gêneros por filme.
        
        \item \textbf{Limpeza de texto de sinopses:} \\
        Remoção de caracteres especiais via Regex, tokenização básica e normalização para \textit{lowercase}. \\
        \textit{Complexidade:} $\mathcal{O}(N \cdot L)$, onde $L$ é o tamanho médio em caracteres das sinopses.
        
        \item \textbf{Exportação:} Persistência do arquivo estruturado \texttt{processed\_movies.csv}. \\
        \textit{Complexidade:} $\mathcal{O}(N)$ de escrita em disco.
    \end{enumerate}
    \vspace{0.1cm}
    \footnotesize{\textit{* O custo total do pipeline é linear $\mathcal{O}(N \cdot L)$, dominado estritamente pelo volume de texto bruto do corpus.}}
\end{frame}

\begin{frame}{6. Resolução de Bug: Compatibilidade de Métricas}
    \framesubtitle{Ajuste crucial em \texttt{generate\_resources.py} para o índice HNSW}
    \begin{itemize}
        \item \textbf{O Problema:} O algoritmo HNSW nativo do FAISS computa distâncias  Euclidianas. Contudo, o escopo de busca semântica exige a métrica de \textbf{Similaridade de Cosseno}.
        \item \textbf{A Solução Matemática:} Aplicou-se a normalização geométrica de vetores antes da indexação no grafo.
        \end{itemize}
\end{frame}

% ---------------------------------------------------
% Slide 10: Ponto 7 - Treinamento
% ---------------------------------------------------
\begin{frame}{7. Treinamento e Indexação dos Modelos}
    \framesubtitle{Configurações e Complexidades Teóricas}
    \begin{itemize}
        \item \textbf{Word2Vec (Gensim):} 
        \begin{itemize}
            \item Treinado do zero sobre o corpus. Parâmetros: \texttt{vector\_size=100}, \texttt{window=5}, \texttt{min\_count=1}.
            \item \textit{Complexidade de Treino:} $\mathcal{O}(E \cdot W \cdot V)$, onde $E$ é o número de épocas, $W$ janelas e $V$ vocabulário.
        \end{itemize}
        \vspace{0.1cm}
        \item \textbf{Sentence Embeddings (all-MiniLM-L6-v2):}
        \begin{itemize}
            \item Modelo baseado em Transformer pré-treinado. Geração de vetores densos de 384 dimensões.
            \item \textit{Complexidade de Extração:} $\mathcal{O}(N \cdot L^2)$ devido à atenção quadrática do Transformer.
        \end{itemize}
        \vspace{0.1cm}
        \item \textbf{Grafo HNSW (FAISS):}
        \begin{itemize}
            \item Construído sobre as matrizes densas do MiniLM. Configurado com parâmetro de conectividade $M=32$.
            \item \textit{Complexidade de Indexação:} $\mathcal{O}(N \log N)$, gerando buscas de latência sub-linear.
        \end{itemize}
    \end{itemize}
\end{frame}

% ---------------------------------------------------
% Slide 11: Ponto 8 - Formação da resposta
% ---------------------------------------------------
\begin{frame}{8. Como o servidor forma a resposta}
    \framesubtitle{Estratégia de RAG Determinística e Resiliente}
    \begin{itemize}
        \item \textbf{Injeção Estrita de Contexto (RAG):}
        \begin{itemize}
            \item Apenas os metadados e os resumos dos filmes recuperados no \texttt{top\_k} da busca alimentam o prompt. O corpus inteiro jamais satura a janela de tokens.
        \end{itemize}
        \vspace{0.1cm}
        \item \textbf{Configuração da Janela do LLM (Ollama):}
        \begin{itemize}
            \item Modelo utilizado localmente: \textbf{Phi-4 Mini}.
            \item Hiperparâmetros técnicos restritivos: \texttt{temperature=0.1} e \texttt{top\_p=0.3}. Força o modelo a agir como um formatador factual, minimizando alucinações de títulos de filmes.
        \end{itemize}
        \vspace{0.1cm}
        \item \textbf{Mecanismo de Resiliência (Fallback):}
        \begin{itemize}
            \item Implementada a classe de exceção personalizada \texttt{LLMResponseError} em \texttt{phi4\_mini.py}. Caso a resposta do modelo falhe nas checagens de sanidade ou o Ollama caia, o sistema chaveia para a exibição direta e estruturada dos metadados extraídos da busca semântica, sem quebrar o app.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{9 e 10. Recursos Computacionais e Tempos de Execução}
    \begin{itemize}
        \item \textbf{Recursos Disponíveis para o Grupo:} Notebook Intel Core i7-11800H (8 núcleos / 16 threads), 16 GB RAM, GPU RTX 3050 Ti. Toda a indexação matemática e inferência corre em CPU (\texttt{faiss-cpu}), provando a viabilidade local da arquitetura RAG sem infraestrutura cara.
        \item \textbf{Tempos Medidos de Execução por Fase:}
        \begin{itemize}
            \item \textbf{Pré-processamento (\texttt{clean\_data.py}):} $\sim$ 2 segundos.
            \item \textbf{Extração de Embeddings (Transformer MiniLM):} $\sim$ 73 segundos.
            \item \textbf{Treinamento Word2Vec do zero + Vetores Médios:} $\sim$ 30 segundos.
            \item \textbf{Construção do Índice HNSW (FAISS):} $\sim$ 2 segundos.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Benchmark: Comparação Teórica e Prática dos 4 Métodos}
    \framesubtitle{Métricas extraídas sobre 55 queries controladas com gabarito histórico}
    \begin{table}[]
        \centering
        \small
        \begin{tabular}{@{}l c r r c c@{}}
            \toprule
            \textbf{Método de Busca} & \textbf{Complexidade Teórica} & \textbf{Construção} & \textbf{Busca (ms)} & \textbf{hit@5} & \textbf{MRR} \\
            \midrule
            TF-IDF + Cosseno & $\mathcal{O}(N \cdot V_{\text{query}})$ & 3,6s & $\sim$78 ms & 0,16 & 0,12 \\
            Word2Vec Average & $\mathcal{O}(N \cdot d)$ & 30,0s & $\sim$10 ms & 0,00 & 0,00 \\
            Sentence Emb. (Linear) & $\mathcal{O}(N \cdot d)$ & 73,0s & $\sim$65 ms & \textbf{0,35} & \textbf{0,25} \\
            Sentence Emb. (HNSW) & $\mathcal{O}(\log N \cdot d)$ & 75,0s & \textbf{$\sim$6 ms} & 0,27 & 0,22 \\
            \bottomrule
        \end{tabular}
    \end{table}
    \vspace{0.1cm}
    \begin{itemize}
        \item \textbf{Análise Teórica vs Prática:} O HNSW atinge o cenário ideal. Sendo matematicamente sub-linear ($\mathcal{O}(\log N)$), ele é \textbf{11 vezes mais rápido} que a busca linear exata do Transformer, sacrificando uma margem aceitável de qualidade (ajuste fino de trade-off aproximado).
        \item \textbf{A Falha do Word2Vec:} Embora rápido (10ms), a média simples de vetores dilui as sinopses longas, resultando em métricas nulas (0,00) de acerto.
    \end{itemize}
\end{frame}

\begin{frame}{11 e 12. Trade-offs: Tempo x Qualidade}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{Tempo de Treino e Indexação x Qualidade}
            \begin{itemize}
                \item Mudar para modelos com maior dimensionalidade (ex: \texttt{all-mpnet-base-v2}, 768d) eleva a precisão, mas dobra o tempo de geração de recursos e a alocação de memória RAM.
            \end{itemize}
            \vspace{0.3cm}
            \textbf{Tempo de Inferência x Qualidade}
            \begin{itemize}
                \item Expandir o \texttt{top\_k} da busca recupera sinopses raras, contudo infla o número de tokens enviados ao prompt. A latência de inferência do LLM local cresce linearmente com o tamanho do contexto injetado.
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \textbf{Análise por Perfil de Query} \\
            \footnotesize (Métrica hit@5 usando Sentence Embeddings) \normalsize
            \begin{itemize}
                \item Queries detalhadas/técnicas: \textbf{0,35 a 0,47}
                \item Queries curtas/vagas de usuários: \textbf{0,12}
            \end{itemize}
            \vspace{0.2cm}
            \small
            As buscas reais de usuários são intrinsicamente mais difíceis devido à falta de palavras-chave, mas a utilidade do sistema escala significativamente com a inclusão do motor semântico em relação aos modelos puramente de frequência de palavras.
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{13. Demais aspectos relevantes}
    \begin{itemize}
        \item \textbf{Limitações Identificadas:}
        \begin{itemize}
            \item O corpus e os modelos operam restritamente na língua inglesa.
            \item Modelos de linguagem pequenos de poucos bilhões de parâmetros (Phi-4 Mini), embora rápidos em CPU, são sensíveis a instruções negativas severas no prompt.
        \end{itemize}
        \vspace{0.2cm}
        \item \textbf{Principais Aprendizados do Grupo:}
        \begin{itemize}
            \item Domínio prático de estruturas de dados avançadas para busca em alta dimensionalidade (grafos de proximidade aproximados HNSW).
            \item Compreensão dos gargalos de tempo de execução (Treinamento/Vetorização Offline vs Inferência Online).
        \end{itemize}
        \vspace{0.2cm}
        \item \textbf{Próximos Passos:}
        \begin{itemize}
            \item Implementação de técnicas de compressão vetorial (Quantização Vetorial) para otimização extrema de memória.
            \item Adição de uma camada de tradução dinâmica para queries em português.
        \end{itemize}
    \end{itemize}
